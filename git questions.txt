git init - создает новый репозиторий гит
git clone - копирует существующий репозиторий
git add - сохраняет текущее состояние проекта
git commit - сохраняет изменения в проекте
git push - отправляет изменения с локального на удаленный репозиторий
git pull - извлекает из удаленного репозитория информацию
git branch - создает новую ветку
git checkout - команда для перемещения по веткам
git merge - сливает коммиты с основной веткой
git rebase - слияние коммитов
git cherry-pick - выборочно можно слить коммиты
git status - посмотреть текущий статус проекта
git log - история проекта
git reset - отменяет изменения
git revert - отменяет изменения и можно создать новые
git stash - сохраняет изменения в локальное хранилище
git diff - разница между изменениями
